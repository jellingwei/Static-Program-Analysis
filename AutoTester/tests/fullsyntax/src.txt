procedure One {
  if x then {       \\1
    while b {       \\2
      b = b + 1;    \\3
    }
    c = 4;          \\4
  } 
  else {
    if y then {     \\5
      a = 5 * c + x * b;        \\6
      b = a + b;                \\7
      while a {                 \\8
        call Two;               \\9
        if c then {             \\10
          c = c * (x - ((y + z) * a) + b);      \\11
        } else {
          a = b;                                \\12
          b = (z + (y - x)) * (a + b * c );     \\13
        }
      }
      z = 5;                   \\14
    } 
    else {
      if z then {             \\15
        call Three;           \\16
        call Four;            \\17
      }
      else {  
        while x {                   \\18
          while y {                 \\19
            a = x * y * (y + x);    \\20
          }
        }
      }
    }
  }
  y = a + b * c;      \\22
  x = x * x;        \\23
}

procedure Two {
  b = a - b;        \\24
  c = 0;            \\25
}

procedure Three {
  while c {       \\26
    call Four;      \\27
    while z {       \\28
      y = x - ((y - y) + z * (a * c));      \\29
    }
    z = x + y + z;      \\30
  }
}

procedure Four {
  if b then {             \\31
    z = z - (b + 1);      \\32
  } else {
    z = a * 10 - b - c;     \\33
  }
}
